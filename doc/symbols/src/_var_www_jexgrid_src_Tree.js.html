265  prototype.traverse = function(args, index) {
266 	 if (args['stop']) {
267 		 return args;
268 	 }
269 	 if (args['up']) {
270 		 if (this.isRoot()) {
271 			 this.callFn(args);
272 		 }
273 		 else if (args['post']) {
274 			 this.parent.traverse(args);
275 			 this.callFn(args);
276 		 }
277 		 else {
278 			 this.callFn(args);
279 			 this.parent.traverse(args);
280 		 }
281 	 }
282 	 else {
283 		 var i = 0,
284 			 c = this.children,
285 			 len = c.length;
286 		 if (args['post']) {
287 			 for (; i < len; i++) {
288 				 c[i].traverse(args, i);
289 			 }
290 			 this.callFn(args, index);
291 		 }
292 		 else {
293 			 this.callFn(args, index);
294 			 if (args['propagate'] === false) {
295 				 delete args['propagate'];
296 			 }
297 			 else {
298 				 for (; !args['stop'] && i < len; i++) {
299 					 c[i].traverse(args, i);
300 				 }
301 			 }
302 		 }
303 	 }
304 	 return args;
305  };
