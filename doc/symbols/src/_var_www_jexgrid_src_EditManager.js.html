556 /**

566 prototype.begin = function(cell) {
567 	if (this.active()) {
568 		this.commit();
569 	}
570 		
571 	if (this.isEditable(cell)) {
572 		var node = cell.getNode();
573 		if (node) {
574 			var editor = this.editor = cell.getColDef().editor;
575 
576 			editor.cell = cell;
577 			editor.grid = this.grid;
578 			editor.before = node.innerHTML;
579 			node.innerHTML = editor.edit(cell.getValue());
580 
581 			cell.get$().addClass(this._options['classEdit']);
582 			editor.focus();
583 		}
584 	}

635 prototype.commit = function() {
636 	if (!this._beingCommitted && this.active()) {
637 		this._beingCommitted = true;
638 
639 		var cell = this.editor.cell;
640 		if (cell) {
641 			var node = cell.getNode();
642 			if (node) {
643 				var value = this.editor.value(node);
644 
645 				if (value == null || value == cell.getValue()) {
646 					this.cancel();
647 				}
648 				else {
649 					var res = cell.setValue(value),
650 						classRes,
651 						$cell = cell.get$();
652 					if (res instanceof Error) {
653 						this.cancel();
654 						classRes = this._options['classFailure'];
655 						$cell.addClass(classRes);
656 						setTimeout(function() {$cell.removeClass(classRes);}, 1000);
657 					}
658 					else {
659 						this._deleteEditor();
660 						this.grid['view'].focus();
661 						this.grid['printMessage']("Successfully Updated.");
662 						classRes = this._options['classSuccess'];
663 						$cell.addClass(classRes);
664 						setTimeout(function() {$cell.removeClass(classRes);}, 1000);
665 					}
666 				}
667 				//$cell.removeClass(this._options['classEdit']);
668 			}
669 		}
670 
671 		this._beingCommitted = false;
672 	}
673 };

684 prototype._deleteContents = function(e, selectionRows, selectionCols) {
685 	if (this.active()) {
686 		return;
687 	}
688 	var updates = {},
689 		fakerows = {},
690 		updatelist = [],
691 		col,
692 		colDef,
693 		defaultValue,
694 		key,
695 		cell,
696 		datarow,
697 		selCol,
698 		row;
699 
700 	col_loop:
701 	for (col in selectionCols) {
702 		if (!selectionCols.hasOwnProperty(col) || col === "length") {
703 			continue;
704 		}
705 
706 		colDef = undefined;
707 		defaultValue = undefined;
708 		key = undefined;
709 		selCol = selectionCols[col];
710 
711 		row_loop:
712 		for (row in selCol) {
713 			if (!selCol.hasOwnProperty(row) || row === "length" || fakerows.hasOwnProperty(row)) {
714 				continue;
715 			}
716 
717 			cell = selCol[row].cell;
718 
719 			if (Util.isNull(colDef)) {
720 				colDef = cell.getColDef();
721 				defaultValue = colDef['defaultValue'];
722 				key = colDef['key'];
723 
724 				if (Util.isNull(colDef['editor'])) {
725 					continue col_loop;
726 				}
727 			}
728 
729 			if (Util.isNotNull(updates[row])) {
730 				datarow = updates[row].datarow;
731 			}
732 			else {
733 				datarow = cell.getDatarow();
734 			}
735 
736 			if (!this.grid['dataMgr'].isReal(datarow)) {
737 				fakerows[row] = true;
738 				continue;
739 			}
740 
741 			if (defaultValue === datarow[key]) {
742 				continue;
743 			}
744 
745 			if (Util.isNull(updates[row])) {
746 				updates[row] = {'datarow':datarow, 'change':{}};
747 				updatelist.push(updates[row]);
748 			}
749 
750 			updates[row].change[key] = defaultValue;
751 		}
752 	}
753 
754 	if (updatelist.length !== 0) {
755 		this.grid['dataMgr'].updateList(updatelist);
756 	}
757 };
