139 prototype.register = function(events, fn, target) {
140 	if (Util.isString(events)) {
141 		this._parseAndAdd(events, fn, target);
142 	}
143 	else {
144 		if (Util.isNotNull(events.e)) {
145 			this._parseAndAdd(events.e, events.f, events.t);
146 		}
147 		else {
148 			var i,
149 				len = events.length,
150 				e;
151 			for (; i < len; i++) {
152 				if (Util.isNotNull(e = events[i])) {
153 					this._parseAndAdd(e.e, e.f, e.t);
154 				}
155 			}
156 		}
157 	}

184 prototype.bind = function(events, fn, target) {
185 	if (Util.isString(events)) {
186 		this._parseAndAdd(events, fn, target);
187 	}
188 	else {
189 		var e;
190 		for (e in events) {
191 			if (events.hasOwnProperty(e)) {
192 				this._parseAndAdd(e, events[e], fn);
193 			}
194 		}
195 	}

198 prototype._parseAndAdd = function(events, fn, target) {
199 	if (Util.isNull(target)) {
200 		target = window;
201 	}
202 
203 	var arr = Util.split(events),
204 		len = arr.length,
205 		i = 0;
206 	if (Util.isFunction(fn)) {	// Function
207 		for (; i < len; i++) {
208 			this._addHandler(arr[i], fn, target);
209 		}
210 	}
211 	else if (Util.isString(fn)) {	// String
212 		var fns = Util.split(fn),
213 			flen = fns.length,
214 			e,
215 			j;
216 
217 		for (; i < len; i++) {
218 			e = arr[i];
219 			for (j = 0; j < flen; j++) {
220 				this._addHandler(e, target[fns[j]], target);
221 			}
222 		}
223 	}
224 	else {	// Function[]
225 		var flen = fn.length,
226 			e,
227 			j;
228 
229 		for (; i < len; i++) {
230 			e = arr[i];
231 			for (j = 0; j < flen; j++) {
232 				this._addHandler(e, fn[j], target);
233 			}
234 		}
235 	}
236 };

239 	if (!this._map.hasOwnProperty(e)) {
240 		this._map[e] = [];
241 	}
242 
243 	this._map[e].push({'fn':fn, 'target':target});

246 /**
247   이벤트 핸들러를 언레지스터 합니다. fn 파라미터를 입력할 경우 fn 핸들러만
248   제거하고 그렇지 않은 경우 이벤트 이름에 레지스터된 모든 핸들러를 제거합니다.
249 
250   @function {} unregister
251   @param {string} event - 이벤트 이름
252   @param {Function=} fn - 이벤트 핸들러 함수
253 
254   @author 조준호
255   @since 1.0.0
256   @version 1.0.0
257   */
258 prototype.unregister = function(event, fn) {
259 	var map = this._map;
260 	if (!map.hasOwnProperty(event)) {
261 		return;
262 	}
263 
264 	var hans = map[event];
265 	if (Util.isNull(fn)) {
266 		hans.length = 0;
267 		delete map[event];
268 		return;
269 	}
270 
271 	var i = 0,
272 		len = hans.length;
273 	for (; i < len; i++) {
274 		if (hans[i].fn === fn) {
275 			hans.removeAt(i);
276 			if (hans.length === 0) {
277 				delete map[event];
278 			}
279 			return;
280 		}
281 	}
282 };

285 /**
286   하나 이상의 이벤트를 트리거 합니다. 해당 이름에 레지스터 된 모든 이벤트 핸들러에게
287   파라미터를 넘겨주고 호출을 한후 결과를 Array 에 넣고 리턴합니다.
288 
289   @function {?[]} trigger
290   @param {string} event - 이벤트 이름. 여러 이벤트의 이름을 빈칸으로 띄우면 여러 이벤트를 동시에 트리거합니다.
291   @param {?[]=} args - 이벤트 핸들러에 입력될 파라미터 어레이
292   @param {Function(?)=} filter - 이벤트 리턴 값 필터링 함수. 리턴
293   값들을 필터링 할 때 사용됩니다. 리턴 값을 파라미터로 받아서 true 를 리턴하면
294   리턴 값은 이벤트 리턴 Array 에 추가됩니다.
295   @returns {?[]} Event Result 어레이
296 
297   @author 조준호
298   @since 1.0.0
299   @version 1.1.7
300   */
301 prototype.trigger = function(e, args, noRes) {
302 	// firing single event
303 	this.grid.log('firing event=' + e, 3);//IF_DEBUG
304 	var map = this._map;
305 	if (map.hasOwnProperty(e)) {
306 		var hans = map[e],
307 			hlen = hans.length;
308 		if (hlen) {
309 			this.grid.log(hlen + ' handlers registered for event=' + e, 4);//IF_DEBUG
310 
311 			var i = 0,
312 				handler;
313 
314 			if (noRes) {
315 				if (args && args.length) {
316 					for (; i < hlen; i++) {
317 						handler = hans[i];
318 						handler.fn.apply(handler.target, args);
319 					}
320 				}
321 				else {
322 					for (; i < hlen; i++) {
323 						handler = hans[i];
324 						handler.fn.call(handler.target);
325 					}
326 				}
327 			}
328 			else {
329 				var rarr = arguments[3] || [];
330 				if (args && args.length) {
331 					for (; i < hlen; i++) {
332 						handler = hans[i];
333 						rarr.push(handler.fn.apply(handler.target, args));
334 					}
335 				}
336 				else {
337 					for (; i < hlen; i++) {
338 						handler = hans[i];
339 						rarr.push(handler.fn.call(handler.target));
340 					}
341 				}
342 				return rarr;
343 			}
344 		}
345 		else {
346 			this.grid.log('no handlers registered for event=' + e, 4);//IF_DEBUG
347 		}
348 	}
349 	else {
350 		this.grid.log('no handlers registered for event=' + e, 4);//IF_DEBUG
351 	}
352 };
