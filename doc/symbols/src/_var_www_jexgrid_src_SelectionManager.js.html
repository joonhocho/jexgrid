317 prototype._mousedownCanvas = function(e, cell) {
318 	if (Util.isNotNull(this._last) && this._last.equals(cell)) {
319 		return;
320 	}
321 	/**
322 	  캔버스에 mousedown 이벤트가 발생하여 해당 셀을 선택하기 전에 발생되는 이벤트 입니다.
323 
324 	  @event {Event} onBeforeSelect
325 	  @param {jQuery.Event} e - jQuery 이벤트 오브젝트
326 	  @param {jx.grid.Cell} cell - 선택될 셀
327 
328 	  @author 조준호
329 	  @since 1.1.7
330 	  @version 1.1.7
331 	  */
332 	this.grid['event'].trigger("onBeforeSelect", [e, cell], true);
333 
334 	if (this._options['multiSelectEnabled'] === false) {
335 		this.selectCell(cell);
336 	}
337 	else {
338 		if (e.shiftKey && Util.isNotNullAnd(this._last, this._range)) {
339 			this.selectRange(cell);
340 		}
341 		else if (e.ctrlKey) {
342 			if (cell.getKey() === this._options['rowSelKey']) {
343 				this.addRow(cell);
344 			}
345 			else {
346 				this.addCell(cell);
347 			}
348 		}
349 		else {
350 			this.selectCell(cell);
351 		}
352 	}

678 prototype._getCellIdxToNavigate = function(dir, row, col) {
679 	switch (dir) {
680 		case this._consts._RIGHT:
681 			if (col < this.grid['colDefMgr'].length() - 1) {
682 				col++;
683 			}
684 			break;
685 		case this._consts._LEFT:
686 			if (col > 0) {
687 				col--;
688 			}
689 			break;
690 		case this._consts._DOWN:
691 			if (row < this.grid['dataMgr'].datalist.length - 1) {
692 				row++;
693 			}
694 			break;
695 		case this._consts._UP:
696 			if (row > 0) {
697 				row--;
698 			}
699 			break;
700 		case this._consts._PGDN:
701 			row += this.grid['view']._options['rowsPerPage'];
702 			if (row > this.grid['dataMgr'].datalist.length - 1) {
703 				row = this.grid['dataMgr'].datalist.length - 1;
704 			}
705 			break;
706 		case this._consts._PGUP:
707 			row -= this.grid['view']._options['rowsPerPage'];
708 			if (row < 0) {
709 				row = 0;
710 			}
711 			break;
712 		case this._consts._HOME:
713 			row = 0;
714 			break;
715 		case this._consts._END:
716 			row = this.grid['dataMgr'].datalist.length - 1;
717 			break;	
718 	}
719 	return [row, col];
720 };
