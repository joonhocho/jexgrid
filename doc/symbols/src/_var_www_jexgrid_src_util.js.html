835 		return true;

839 		type = arch[0];

842 		if (type === "array") {
843 			return Util.areEqualArrays(ar, br);
844 		}
845 		return Util.areEqualObjects(ar, br);
846 	}
847 	if (len > 1) {
848 		var subArch = arch.slice(1),
849 			i = 0;
850 		if (type === "array") {
851 			if (ar.length !== br.length) {
852 				return false;
853 			}
854 			var alen = ar.length;
855 			for (; i < alen; i++) {
856 				if (!br.hasOwnProperty(i) || !Util.areEqualComplex(ar[i], br[i], subArch)) {
857 					return false;
858 				}
859 			}
860 			return true;
861 		}
862 		else {
863 			for (i in ar) {
864 				if (ar.hasOwnProperty(i)) {
865 					if (!br.hasOwnProperty(i) || !Util.areEqualComplex(ar[i], br[i], subArch)) {
866 						return false;
867 					}
868 				}
869 			}
870 			for (i in br) {
871 				if (br.hasOwnProperty(i)) {
872 					if (!ar.hasOwnProperty(i) || !Util.areEqualComplex(ar[i], br[i], subArch)) {
873 						return false;
874 					}
875 				}
876 			}
877 			return true;		
878 		}
879 	}

882 /**
883   Check type of given object against given type, and
884   either return false or throw a 'TypeError' exception if the type is not correct.
885 
886   @function {boolean} typeCheck
887 
888   @param {string | Function} type - correct type of object to check against
889   @param {*} obj - object to check type of
890   @param {boolean=} allowUndefined - if true, undefined for object is considered as correct type
891   @param {boolean=} allowNull - if true, null for object is considered as correct type
892   @param {boolean=} noThrow - if true, return false instead of throwing 'TypeError' exception
893 
894   @return true if type of object is correct
895 
896   @throws TypeError when the type of object is incorrect
897 
898   @author 
899   @since 1.3.0
900   @version 1.3.0
901   */
902 Util.typeCheck = function(type, obj, allowUndefined, allowNull, noThrow) {
903 	// return true when object is undefined and undefined is allowed, or
904 	// object is null and null is allowed
905 	if ((allowUndefined && obj === undefined) || (allowNull && obj === null)) {
906 		return true;
907 	}
908 
909 	switch (typeof type) {
910 		case "string":
911 			// return true if type is correct, or
912 			if (typeof obj === type) {
913 				return true;
914 			}
915 			break;
916 		case "function":
917 			if (obj instanceof type) {
918 				return true;
919 			}
920 			break;
921 	}
922 
923 	// return false instead if noThrow is set
924 	if (noThrow) {
925 		return false;
926 	}
927 
928 	// throw an exception
929 	throw new TypeError("object is not a " + type + ", but is a " + typeof obj);
930 };

935   @function {string} sprint
936 
937   @param {string} str - string with variables
938   @param {Object} obj - object of variable to value mappings
939   @param {string=} prefix - variable prefix. default to '%'.
940   @param {string=} suffix - variable suffix. default to '%'.

944   @throws TypeError when parameter types are incorrect
945 
946   @author

948   @version 1.3.0
949   */
950 Util.sprint = function(str, obj, prefix, suffix) {
951 	// check types of parameters
952 	Util.typeCheck("string", str);
953 	Util.typeCheck("object", obj);
954 	Util.typeCheck("string", prefix, true); // allow undefined
955 	Util.typeCheck("string", suffix, true); // allow undefined
956 
957 	var prefixLength,
958 		suffixLength,
959 		i;
960 
961 	// if prefix is undefined, set '%' as default prefix
962 	if (prefix === undefined) {

964 	}
965 
966 	// if suffix is undefined, set '%' as default suffix
967 	if (suffix === undefined) {
968 		suffix = '%';
969 	}
970 
971 	// iterate through object and replace each key string with value string
972 	for (i in obj) {
973 		if (obj.hasOwnProperty(i)) {
974 			str = str.replace(new RegExp(prefix + i + suffix, "gm"), obj[i]);
975 		}
976 	}
